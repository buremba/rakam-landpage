    Did our Facebook ADS campaign worked well enough to increase our revenue?
    SELECT total_order_amount / fb.spent_amount_total AS revenue_ads_ratio FROM (
        SELECT cast(_time AS date) day, sum(total_amount) AS total_order_amount FROM (
            SELECT _user, total_amount FROM collection.transaction
               JOIN mysqldb.users ON (users.id = transaction._user)
           UNION ALL SELECT user_id, total_amount FROM remotefile.offlinesales
        ) WHERE _user IN (SELECT user_id FROM materialized.gold_users WHERE clearbit_job = 'Doctor')
    ) orders JOIN fb_advertisement fb ON (orders.day = fb.day)
